## Task 1: Error Recovery in `dispatch-agent.sh`

### Steps

The changes add: (a) an `ERR` trap that captures failure output, updates the session to `failed`, sets the agent to `error`, and conditionally resets the task for retry; (b) a configurable timeout (reads `stuck_agent_timeout_minutes` from the `config` table, defaults to 10); (c) background-mode timeout via a watchdog subshell.

```bash
#!/usr/bin/env bash
# dispatch-agent.sh --- Dispatch a Claude CLI agent with layered prompts
#
# Usage:
#   bash scripts/dispatch-agent.sh <agent_id> <task_id> [--model sonnet] [--background]
#
# The script:
# 1. Reads the agent's prompt_file and config from SQLite
# 2. Reads the task context from SQLite
# 3. Creates a session record for tracking
# 4. Builds a consolidated system prompt (base + protocols + browser if needed)
# 5. Dispatches via `claude -p` with --append-system-prompt-file
# 6. Updates session and agent status on completion
# 7. On failure: captures error, marks session failed, retries if allowed

set -uo pipefail

PROJECT_DIR="C:/Users/jfcho/Desktop/CoWork/MultiAgent"
DB_PATH="$PROJECT_DIR/ultra.db"
LOGS_DIR="$PROJECT_DIR/logs"
mkdir -p "$LOGS_DIR"

AGENT_ID="${1:?Usage: dispatch-agent.sh <agent_id> <task_id>}"
TASK_ID="${2:?Usage: dispatch-agent.sh <agent_id> <task_id>}"
MODEL="${3:-sonnet}"
BACKGROUND="${4:-}"

# --- Helper: read timeout from config (default 10 min = 600s) ---
TIMEOUT_SECONDS=$(python -c "
import sqlite3
db = sqlite3.connect('$DB_PATH')
row = db.execute(\"SELECT value FROM config WHERE key = 'stuck_agent_timeout_minutes'\").fetchone()
print(int(row[0]) * 60 if row else 600)
db.close()
")

# --- Error handler ---
handle_failure() {
    local exit_code=$1
    local error_output="${2:-Agent process exited with code $exit_code}"

    echo "[dispatch] FAILURE agent=$AGENT_ID task=$TASK_ID exit=$exit_code" >&2

    python -c "
import sqlite3, json
from datetime import datetime, timezone
db = sqlite3.connect('$DB_PATH')
ts = datetime.now(timezone.utc).strftime('%Y-%m-%dT%H:%M:%SZ')

# Update session to failed
row = db.execute('SELECT started_at FROM sessions WHERE session_id = ?', ('$SESSION_ID',)).fetchone()
duration = 0
if row:
    try:
        s = datetime.fromisoformat(row[0].replace('Z', '+00:00'))
        e = datetime.fromisoformat(ts.replace('Z', '+00:00'))
        duration = (e - s).total_seconds()
    except:
        pass
error_msg = '''$(echo "$error_output" | head -c 5000 | sed "s/'/\\\\'/g")'''
db.execute('UPDATE sessions SET status = ?, completed_at = ?, duration_seconds = ?, success = 0, error_message = ? WHERE session_id = ?',
    ('failed', ts, duration, error_msg, '$SESSION_ID'))

# Update agent to error
db.execute('UPDATE agents SET status = ?, error_count = error_count + 1, updated_at = ? WHERE agent_id = ?',
    ('error', ts, '$AGENT_ID'))

# Update task: increment retry_count, check if retryable
task = db.execute('SELECT retry_count, max_retries FROM tasks WHERE task_id = ?', ('$TASK_ID',)).fetchone()
if task:
    new_retry = task[0] + 1
    if new_retry < task[1]:
        # Reset to pending for re-dispatch
        db.execute('UPDATE tasks SET status = ?, retry_count = ?, error_message = ?, updated_at = ? WHERE task_id = ?',
            ('pending', new_retry, error_msg, ts, '$TASK_ID'))
        print(f'[dispatch] Task $TASK_ID queued for retry ({new_retry}/{task[1]})')
    else:
        # Max retries exceeded
        db.execute('UPDATE tasks SET status = ?, retry_count = ?, error_message = ?, updated_at = ? WHERE task_id = ?',
            ('failed', new_retry, error_msg, ts, '$TASK_ID'))
        print(f'[dispatch] Task $TASK_ID FAILED permanently ({new_retry}/{task[1]} retries exhausted)')

db.commit()
db.close()
"
}

# --- 1. Read agent info from DB ---

import sqlite3, json
db = sqlite3.connect('$DB_PATH')
db.row_factory = sqlite3.Row
row = db.execute('SELECT * FROM agents WHERE agent_id = ?', ('$AGENT_ID',)).fetchone()

db.commit()

    pass

Task ID: $TASK_ID
$(echo "$TASK_JSON" | python -c "

    if [ -f "$PROJECT_DIR/prompts/$protocol" ]; then
        echo "" >> "$CONSOLIDATED"

    config = json.loads(sys.stdin.read())
    print(config.get('browser_category', ''))
except:
    print('')
")

if [ -n "$BROWSER_CAT" ]; then
    if [ -f "$PROJECT_DIR/prompts/browser-protocol.md" ]; then
        echo "" >> "$CONSOLIDATED"
        cat "$PROJECT_DIR/prompts/browser-protocol.md" >> "$CONSOLIDATED"
        echo "[dispatch] Browser protocol appended (category=$BROWSER_CAT)" >&2
    fi
fi

# --- 6. Dispatch ---
echo "[dispatch] Agent=$AGENT_ID Task=$TASK_ID Model=$MODEL Timeout=${TIMEOUT_SECONDS}s" >&2

CLAUDE_CMD="claude -p --model $MODEL --output-format json --dangerously-skip-permissions"
CLAUDE_CMD="$CLAUDE_CMD --append-system-prompt-file $CONSOLIDATED"
CLAUDE_CMD="$CLAUDE_CMD --add-dir $PROJECT_DIR"

if [ "$BACKGROUND" = "--background" ]; then
    # Background dispatch with timeout watchdog
    $CLAUDE_CMD "$TASK_PROMPT" > "$LOGS_DIR/$AGENT_ID-$TASK_ID.json" 2>&1 &
    PID=$!
    echo "[dispatch] Background PID=$PID" >&2

    # Store PID for health monitoring
    python -c "
import sqlite3
db = sqlite3.connect('$DB_PATH')
db.execute('UPDATE agents SET session_id = ? WHERE agent_id = ?', ('$PID', '$AGENT_ID'))
db.commit()
db.close()
"

    # Watchdog: kill agent if it exceeds timeout
    (
        sleep "$TIMEOUT_SECONDS"
        if kill -0 "$PID" 2>/dev/null; then
            echo "[dispatch] TIMEOUT: Killing agent $AGENT_ID (PID=$PID) after ${TIMEOUT_SECONDS}s" >&2
            kill "$PID" 2>/dev/null
            sleep 2
            kill -9 "$PID" 2>/dev/null

            python -c "
import sqlite3
from datetime import datetime, timezone
db = sqlite3.connect('$DB_PATH')
ts = datetime.now(timezone.utc).strftime('%Y-%m-%dT%H:%M:%SZ')
db.execute('UPDATE sessions SET status = ?, completed_at = ?, error_message = ? WHERE session_id = ?',
    ('timeout', ts, 'Agent exceeded timeout of ${TIMEOUT_SECONDS}s', '$SESSION_ID'))
db.execute('UPDATE agents SET status = ?, error_count = error_count + 1, updated_at = ? WHERE agent_id = ?',
    ('error', ts, '$AGENT_ID'))
task = db.execute('SELECT retry_count, max_retries FROM tasks WHERE task_id = ?', ('$TASK_ID',)).fetchone()
if task:
    new_retry = task[0] + 1
    if new_retry < task[1]:
        db.execute('UPDATE tasks SET status = ?, retry_count = ?, error_message = ?, updated_at = ? WHERE task_id = ?',
            ('pending', new_retry, 'Timeout after ${TIMEOUT_SECONDS}s', ts, '$TASK_ID'))
    else:
        db.execute('UPDATE tasks SET status = ?, retry_count = ?, error_message = ?, updated_at = ? WHERE task_id = ?',
            ('failed', new_retry, 'Timeout after ${TIMEOUT_SECONDS}s (retries exhausted)', ts, '$TASK_ID'))
db.commit()
db.close()
"
        fi
    ) &
    WATCHDOG_PID=$!

    echo "{\"pid\": $PID, \"watchdog_pid\": $WATCHDOG_PID, \"agent_id\": \"$AGENT_ID\", \"task_id\": \"$TASK_ID\", \"session_id\": \"$SESSION_ID\", \"timeout_seconds\": $TIMEOUT_SECONDS}"
else
    # Foreground dispatch with timeout
    STDERR_FILE="$LOGS_DIR/$AGENT_ID-$TASK_ID.stderr"
    RESULT=""
    EXIT_CODE=0

    if command -v timeout &>/dev/null; then
        RESULT=$(timeout "$TIMEOUT_SECONDS" $CLAUDE_CMD "$TASK_PROMPT" 2>"$STDERR_FILE") || EXIT_CODE=$?
    else
        RESULT=$($CLAUDE_CMD "$TASK_PROMPT" 2>"$STDERR_FILE") || EXIT_CODE=$?
    fi

    # Clean up consolidated prompt
    rm -f "$CONSOLIDATED"

    if [ $EXIT_CODE -eq 124 ]; then
        # Timeout exit code
        handle_failure 124 "Agent exceeded timeout of ${TIMEOUT_SECONDS}s"
        # Also update session status specifically to 'timeout'
        python -c "
import sqlite3
from datetime import datetime, timezone
db = sqlite3.connect('$DB_PATH')
ts = datetime.now(timezone.utc).strftime('%Y-%m-%dT%H:%M:%SZ')
db.execute('UPDATE sessions SET status = ? WHERE session_id = ?', ('timeout', '$SESSION_ID'))
db.commit()
db.close()
"
        exit 124
    elif [ $EXIT_CODE -ne 0 ]; then
        ERROR_OUTPUT=$(cat "$STDERR_FILE" 2>/dev/null | head -c 5000)
        handle_failure $EXIT_CODE "$ERROR_OUTPUT"
        exit $EXIT_CODE
    fi

    echo "$RESULT"

    # Update session with results
    python -c "
import sqlite3
from datetime import datetime, timezone
db = sqlite3.connect('$DB_PATH')
ts = datetime.now(timezone.utc).strftime('%Y-%m-%dT%H:%M:%SZ')
row = db.execute('SELECT started_at FROM sessions WHERE session_id = ?', ('$SESSION_ID',)).fetchone()
if row:
    started = row[0]
    try:
        s = datetime.fromisoformat(started.replace('Z', '+00:00'))
        e = datetime.fromisoformat(ts.replace('Z', '+00:00'))
        duration = (e - s).total_seconds()
    except:
        duration = 0
    db.execute('UPDATE sessions SET status = ?, completed_at = ?, duration_seconds = ?, success = ?, output_snapshot = ? WHERE session_id = ?',
        ('completed', ts, duration, 1, '''$(echo "$RESULT" | head -c 10000)''', '$SESSION_ID'))
db.commit()
db.close()
"

    # Mark agent as idle, increment run_count
    python -c "
import sqlite3
from datetime import datetime, timezone
db = sqlite3.connect('$DB_PATH')
ts = datetime.now(timezone.utc).strftime('%Y-%m-%dT%H:%M:%SZ')
db.execute('UPDATE agents SET status = ?, run_count = run_count + 1, updated_at = ? WHERE agent_id = ?',
    ('idle', ts, '$AGENT_ID'))
db.commit()
db.close()
"
fi

passed = 0

errors = []


def test(name, condition, detail=""):
    global passed, failed
    if condition:
        passed += 1
        print(f"  PASS  {name}")
    else:
        failed += 1
        msg = f"  FAIL  {name}"
        if detail:
            msg += f" -- {detail}"
        print(msg)
        errors.append(name)

    """Run db-utils.py with the test database."""
    env = os.environ.copy()
    cmd = [sys.executable, DB_UTILS] + list(args)
    result = subprocess.run(cmd, capture_output=True, text=True, cwd=PROJECT_DIR,
                            env=env, timeout=30)
    return result


def run_db_init():
    """Run db-init.py targeting the test database."""
    result = subprocess.run(
        [sys.executable, DB_INIT],

## Work Release Flow

1. An agent creates a `work_releases` entry with `status = 'pending'`
2. The system checks `auto_release_rules` for a matching rule
   - Match criteria: `match_agent_type`, `match_action_type`, `match_skill_id`, `match_title_pattern`
   - If matched and `is_enabled = 1`: auto-release (set `status = 'auto_released'`, link `auto_release_rule_id`)
3. If no auto-release: the Work Release UI (`ui/`) shows pending items for human review
4. Human approves (`approved`) or rejects (`rejected`)
5. On approval, the agent's task proceeds; on rejection, the task is blocked or revised

## Cron Manager

The cron system uses the `cron_schedule` table:
- Each row maps an `agent_id` to a `cron_expression` (standard 5-field cron syntax)
- `is_enabled` controls whether the schedule is active
- `fire_count` and `max_fires` control bounded execution
- `last_fired_at` and `next_fire_at` track timing
- A 1-minute tick process reads due schedules and dispatches agents via `dispatch-agent.sh`

### Agent stuck in "running" status

# Check how long the agent has been running
python scripts/db-utils.py get-agent <agent_id>

python scripts/db-utils.py query "UPDATE agents SET status = 'idle', updated_at = datetime('now') WHERE agent_id = '<agent_id>'"
```

### Task stuck in "in_progress"
```bash
# Check the session for this task
python scripts/db-utils.py list-sessions --task <task_id>
# Reset task to pending for retry
python scripts/db-utils.py update-task <task_id> --status pending
```

### Database locked errors
The database uses WAL mode for concurrent access. If you see lock errors:
```bash
# Check for zombie processes
python scripts/db-utils.py query "SELECT agent_id, status, session_id FROM agents WHERE status = 'running'"
# Reset all stuck agents
python scripts/db-utils.py query "UPDATE agents SET status = 'idle' WHERE status = 'running'"
```

### Re-initialize database (preserves existing data)
```bash
python scripts/db-init.py    # safe to re-run, uses IF NOT EXISTS / OR IGNORE
```

### View recent errors
```bash
python scripts/db-utils.py list-sessions --success 0 --limit 10
python scripts/db-utils.py query "SELECT * FROM improvement_log WHERE category = 'failure_pattern' ORDER BY created_at DESC LIMIT 10"
```
